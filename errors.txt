generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum Role {
  STUDENT
  TEACHER
  ADMIN
}

enum LessonType {
  VIDEO
  QUIZ
  DOCUMENT
}

enum QuestionType {
  MCQ
  TRUE_FALSE
  NUMBER
}

model User {
  id        String   @id @default(cuid())
  name      String?
  email     String   @unique
  password  String
  role      Role     @default(STUDENT)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  teacherCourses  Course[]          @relation("TeacherCourses")
  enrollments     Enrollment[]
  codesUsed       Code[]            @relation("UserCodes")
  progresses      Progress[]
  watchProgresses WatchProgress[]
  videoBookmarks  VideoBookmark[]
  quizAnswers     QuizAnswer[]
  contestEntries  ContestEntry[]
  notifications   Notification[]
  payments        Payment[]
  certificates    Certificate[]
  auditLogs       AuditLog[]        @relation("Actor")
  referralsMade   Referral[]        @relation("Referrer")
  refereeOf       Referral?         @relation("Referee")
  questionBank    QuestionBank[]
}

model Course {
  id          String   @id @default(cuid())
  title       String
  description String?
  thumbnail   String?
  price       Float?
  isFree      Boolean  @default(false)
  teacherId   String
  teacher     User     @relation("TeacherCourses", fields: [teacherId], references: [id])
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  modules      Module[]
  enrollments  Enrollment[]
  codes        Code[]
  payments     Payment[]
  certificates Certificate[]
  coupons      Coupon[]
  contests     Contest[]
  questionBank QuestionBank[]
  attachments  Attachment[]
  progresses   Progress[]
}

model Module {
  id       String   @id @default(cuid())
  title    String
  order    Int
  courseId String
  course   Course   @relation(fields: [courseId], references: [id], onDelete: Cascade)

  lessons     Lesson[]
  attachments Attachment[]
}

model Lesson {
  id         String     @id @default(cuid())
  title      String
  videoId    String?    // Bunny Stream GUID
  duration   Int?       // seconds
  order      Int
  type       LessonType @default(VIDEO)
  lockConfig Json?      // إعدادات القفل
  moduleId   String
  module     Module     @relation(fields: [moduleId], references: [id], onDelete: Cascade)

  quiz           Quiz?
  documents      Document[]
  progresses     Progress[]
  watchProgresses WatchProgress[]
  videoBookmarks VideoBookmark[]
  attachments    Attachment[]
}

model Quiz {
  id        String     @id @default(cuid())
  lessonId  String     @unique
  lesson    Lesson     @relation(fields: [lessonId], references: [id], onDelete: Cascade)
  questions QuizQuestion[]
}

model QuizQuestion {
  id       String       @id @default(cuid())
  text     String
  type     QuestionType
  options  String[]     // for MCQ
  answer   String       // correct answer (index, "true"/"false", number string)
  quizId   String
  quiz     Quiz         @relation(fields: [quizId], references: [id], onDelete: Cascade)
  answers  QuizAnswer[]
}

model QuizAnswer {
  id         String       @id @default(cuid())
  userId     String
  user       User         @relation(fields: [userId], references: [id])
  questionId String
  question   QuizQuestion @relation(fields: [questionId], references: [id])
  answer     String
  isCorrect  Boolean?
  createdAt  DateTime     @default(now())
}

model Code {
  id          String    @id @default(cuid())
  code        String    @unique
  courseId    String
  course      Course    @relation(fields: [courseId], references: [id])
  isUsed      Boolean   @default(false)
  userId      String?
  usedBy      User?     @relation("UserCodes", fields: [userId], references: [id])
  purchasedAt DateTime?
  createdAt   DateTime  @default(now())
}

model Coupon {
  id            String   @id @default(cuid())
  code          String   @unique
  discountType  String   // PERCENTAGE, FIXED
  discountValue Float
  maxUses       Int      @default(1)
  currentUses   Int      @default(0)
  minPurchase   Float?
  validFrom     DateTime @default(now())
  validUntil    DateTime
  courseId      String?
  course        Course?  @relation(fields: [courseId], references: [id])
  createdBy     String
  isActive      Boolean  @default(true)
}

model Enrollment {
  id        String    @id @default(cuid())
  userId    String
  user      User      @relation(fields: [userId], references: [id])
  courseId  String
  course    Course    @relation(fields: [courseId], references: [id])
  createdAt DateTime  @default(now())
  expiresAt DateTime? // صلاحية الوصول

  @@unique([userId, courseId])
  @@index([userId])
  @@index([courseId])
}

model Progress {
  id          String    @id @default(cuid())
  userId      String
  user        User      @relation(fields: [userId], references: [id])
  lessonId    String
  lesson      Lesson    @relation(fields: [lessonId], references: [id])
  courseId    String
  course      Course    @relation(fields: [courseId], references: [id])
  completed   Boolean   @default(false)
  completedAt DateTime?

  @@unique([userId, lessonId])
}

model WatchProgress {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id])
  lessonId  String
  lesson    Lesson   @relation(fields: [lessonId], references: [id])
  seconds   Int      @default(0)
  updatedAt DateTime @updatedAt

  @@unique([userId, lessonId])
}

model Document {
  id       String @id @default(cuid())
  url      String
  lessonId String
  lesson   Lesson @relation(fields: [lessonId], references: [id], onDelete: Cascade)
}

model VideoBookmark {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id])
  lessonId  String
  lesson    Lesson   @relation(fields: [lessonId], references: [id])
  timestamp Int
  note      String?
  createdAt DateTime @default(now())

  @@index([userId, lessonId])
}

model Notification {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id])
  message   String
  read      Boolean  @default(false)
  createdAt DateTime @default(now())
}

model Payment {
  id            String   @id @default(cuid())
  userId        String
  user          User     @relation(fields: [userId], references: [id])
  courseId      String
  course        Course   @relation(fields: [courseId], references: [id])
  amount        Float
  status        String   @default("PENDING") // PENDING, COMPLETED, FAILED
  paymobOrderId String?  @unique
  createdAt     DateTime @default(now())
}

model Attachment {
  id         String   @id @default(cuid())
  name       String
  url        String
  type       String   // PDF, IMAGE, DOC
  size       Int
  lessonId   String?
  lesson     Lesson?  @relation(fields: [lessonId], references: [id])
  moduleId   String?
  module     Module?  @relation(fields: [moduleId], references: [id])
  courseId   String?
  course     Course?  @relation(fields: [courseId], references: [id])
  uploadedBy String
  createdAt  DateTime @default(now())
}

model Certificate {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id])
  courseId    String
  course      Course   @relation(fields: [courseId], references: [id])
  issueDate   DateTime @default(now())
  code        String   @unique
  downloadUrl String?

  @@index([userId])
  @@index([courseId])
}

model QuestionBank {
  id         String       @id @default(cuid())
  text       String
  type       QuestionType
  options    String[]
  answer     String
  tags       String[]
  difficulty Int          @default(1)
  teacherId  String
  teacher    User         @relation(fields: [teacherId], references: [id])
  courseId   String?
  course     Course?      @relation(fields: [courseId], references: [id])
  createdAt  DateTime     @default(now())

  @@index([teacherId])
  @@index([courseId])
  @@index([tags])
}

model Contest {
  id        String   @id @default(cuid())
  title     String
  courseId  String
  course    Course   @relation(fields: [courseId], references: [id])
  startDate DateTime
  endDate   DateTime
  type      String   @default("DAILY") // DAILY, WEEKLY, SPECIAL
  questions String[] // QuestionBank IDs
  rewards   Json?
  status    String   @default("ACTIVE") // ACTIVE, ENDED
  createdAt DateTime @default(now())

  entries   ContestEntry[]
}

model ContestEntry {
  id          String    @id @default(cuid())
  contestId   String
  contest     Contest   @relation(fields: [contestId], references: [id])
  userId      String
  user        User      @relation(fields: [userId], references: [id])
  score       Int
  answers     Json
  startedAt   DateTime  @default(now())
  submittedAt DateTime?
  rank        Int?

  @@unique([contestId, userId])
}

model Referral {
  id          String   @id @default(cuid())
  referrerId  String
  referrer    User     @relation("Referrer", fields: [referrerId], references: [id])
  refereeId   String   @unique
  referee     User     @relation("Referee", fields: [refereeId], references: [id])
  rewardGiven Boolean  @default(false)
  createdAt   DateTime @default(now())

  @@index([referrerId])
}

model AuditLog {
  id         String   @id @default(cuid())
  actorId    String?
  actor      User?    @relation("Actor", fields: [actorId], references: [id])
  action     String
  entityType String
  entityId   String
  metadata   Json?
  createdAt  DateTime @default(now())

  @@index([actorId])
  @@index([entityType, entityId])
}

model CourseStats {
  courseId    String   @id
  enrollments Int      @default(0)
  watchTime   Int      @default(0) // seconds
  completions Int      @default(0)
  updatedAt   DateTime @updatedAt
}
